  核心流程                                                                                                                                                                          
                                                                                                                                                                                    
  1. 请求解析 (protocol.py):                                                                                                                                                        
    - ChatCompletionRequest 解析 tools 和 tool_choice 参数
    - tools 被解析为 list[ChatCompletionToolsParam]                                                                                                                                 
  2. 预处理 (serving.py):                                                                                                                                                           
    - render_chat_request 将 tools 转换为 dict 格式                                                                                                                                 
    - _preprocess_chat 将 tools 合并到 chat_template_kwargs                                                                                                                         
  3. Prompt 渲染 (hf.py):                                                                                                                                                           
    - HfRenderer.render_messages 调用 safe_apply_chat_template                                                                                                                      
    - tools 通过 tokenizer.apply_chat_template 传递给 Jinja2 模板                                                                                                                   
  4. Template 渲染 (.jinja 文件):                                                                                                                                                   
    - Jinja2 模板将 tools 渲染为模型可理解的 prompt 格式                                                                                                                            
    - 不同模型有不同的模板格式（Hermes、Llama、Mistral 等）                                                                                                                         
  5. Tokenization:                                                                                                                                                                  
    - 渲染后的 prompt 文本被 tokenizer 编码为 token ids                                                                                                                             
    - token ids 作为模型输入进行推理                                                                                                                                                
  6. 输出解析 (tool_parsers/):                                                                                                                                                      
    - ToolParser 从模型输出中提取 tool calls                                                                                                                                        
    - 支持 extract_tool_calls (非流式) 和 extract_tool_calls_streaming (流式)                                                                                                       
                                                                                                                                                                                    
  关键代码路径                                                                                                                                                                      
                                                                                                                                                                                    
  ChatCompletionRequest → render_chat_request → _preprocess_chat                                                                                                                    
      → ChatParams(tools=...) → HfRenderer.render_messages                                                                                                                          
      → safe_apply_chat_template → tokenizer.apply_chat_template                                                                                                                    
      → Jinja2 template 渲染 → tokenizer.encode → token ids                                                                                                                         
      → EngineClient.generate → 模型输出                                                                                                                                            
      → ToolParser.extract_tool_calls → OpenAI 响应格式

 async def render_chat_request(self, request: ChatCompletionRequest):
     # 1. 将 tools 转换为 dict 格式
     if request.tools is None or (request.tool_choice == "none" and self.exclude_tools_when_tool_choice_none):
         tool_dicts = None
     else:
         tool_dicts = [tool.model_dump() for tool in request.tools]

     # 2. 调用 _preprocess_chat 进行预处理
     conversation, engine_prompts = await self._preprocess_chat(
         request,
         request.messages,
         default_template=self.chat_template,
         default_template_content_format=self.chat_template_content_format,
         default_template_kwargs=self.default_chat_template_kwargs,
         tool_dicts=tool_dicts,
         tool_parser=tool_parser,
     )

 3. Chat 参数构建

 文件: vllm/entrypoints/openai/engine/serving.py

 async def _preprocess_chat(
     self,
     request: RendererChatRequest,
     messages: list[ChatCompletionMessageParam],
     ...
     tool_dicts: list[dict[str, Any]] | None = None,
     tool_parser: Callable[[TokenizerLike], ToolParser] | None = None,
 ):
     # 1. 合并 template kwargs，关键：tools 被加入 kwargs
     default_template_kwargs = merge_kwargs(
         default_template_kwargs,
         dict(
             tools=tool_dicts,  # tools 传递给 chat template
             tokenize=isinstance(renderer.tokenizer, MistralTokenizer),
         ),
     )

     # 2. 构建 ChatParams
     chat_params = request.build_chat_params(
         default_template, default_template_content_format
     ).with_defaults(default_template_kwargs)

     # 3. 渲染消息
     (conversation,), (engine_prompt,) = await renderer.render_chat_async(
         [messages],
         chat_params,
         tok_params,
     )

     # 4. Tool parser 调整请求（如设置 structured outputs）
     if tool_parser is not None:
         tool_choice = getattr(request, "tool_choice", "none")
         if tool_choice != "none":
             request = tool_parser(tokenizer).adjust_request(request=request)

 当 tokenize=True 时，apply_chat_template 内部调用：
 def apply_chat_template(self, conversation, tools=None, ...):
     # 1. 渲染 Jinja2 template 得到 prompt 文本
     rendered_prompt = self.render_template(chat_template, {
         "messages": conversation,
         "tools": tools,
         ...
     })

     # 2. tokenize 得到 token ids
     if tokenize:
         return self.encode(rendered_prompt, add_special_tokens=False)

     return rendered_prompt

 文件: vllm/tool_parsers/abstract_tool_parser.py

 class ToolParser:
     def adjust_request(self, request: ChatCompletionRequest) -> ChatCompletionRequest:
         """为 tool calling 设置 structured outputs"""
         if not request.tools:
             return request

         json_schema_from_tool = get_json_schema_from_tools(
             tool_choice=request.tool_choice,
             tools=request.tools
         )

         if json_schema_from_tool is not None:
             # 设置 JSON schema 约束输出格式
             request.structured_outputs = StructuredOutputsParams(
                 json=json_schema_from_tool
             )
         return request

     def extract_tool_calls(self, model_output: str, request: ChatCompletionRequest):
         """从模型输出中提取 tool calls"""
         raise NotImplementedError

     def extract_tool_calls_streaming(self, ...):
         """流式输出中提取 tool calls"""
         raise NotImplementedError

 文件: vllm/tool_parsers/hermes_tool_parser.py

 class Hermes2ProToolParser(ToolParser):
     def __init__(self, tokenizer: TokenizerLike):
         self.tool_call_start_token = "<tool_call"
         self.tool_call_end_token = "</tool_call"
         # 编码特殊 token
         self.tool_call_start_token_ids = self.model_tokenizer.encode(
             self.tool_call_start_token, add_special_tokens=False
         )

     def extract_tool_calls(self, model_output: str, request):
         """解析 <tool_call... 格式的输出"""
         if self.tool_call_start_token not in model_output:
             return ExtractedToolCallInformation(
                 tools_called=False, tool_calls=[], content=model_output
             )

         # 正则匹配 tool call
         function_call_tuples = self.tool_call_regex.findall(model_output)

         # 解析 JSON 并构建 ToolCall 对象
         raw_function_calls = [json.loads(match[0] if match[0] else match[1])
                              for match in function_call_tuples]

         tool_calls = [
             ToolCall(
                 type="function",
                 function=FunctionCall(
                     name=function_call["name"],
                     arguments=json.dumps(function_call["arguments"]),
                 ),
             )
             for function_call in raw_function_calls
         ]
         return ExtractedToolCallInformation(
             tools_called=True,
             tool_calls=tool_calls,
             content=content,
         )
